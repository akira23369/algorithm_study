#include<iostream>
#include<algorithm>
const int N = 50010;
int n;
std::pair<int, int> l[N];
int main()
{
    std::cin >> n;
    for (int i = 0; i < n; i++)
    {
        int w, s;
        std::cin >> w >> s;
        l[i].first = w + s;
        l[i].second = w;
    }
    sort(l, l + n);
    int sum = 0;
    int res = -2e9;
    for (int i = 0; i < n; i++)
    {
        int s = l[i].first - l[i].second, w = l[i].second;
        res = std::max(res, sum - s);
        sum += w;
    }
    std::cout << res;
}
